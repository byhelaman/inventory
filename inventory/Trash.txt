private void ChangeButtonIcon(Button button, bool isPanelVisible, Image iconUp, Image iconDown)
{

    // Cambia la imagen del botón según la visibilidad del panel
    if (isPanelVisible)
    {
        button.Image = iconUp;
    }
    else
    {
        button.Image = iconDown;
    }
}
